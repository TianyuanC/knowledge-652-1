```javascript Array.prototype.Reduce
const counts = [1, 2, 3];
console.log(`1 + 2 + 3 = ${1 + 2 + 3}`);
// 6

const sum = counts.reduce((acc, count) => acc + count);
console.log(sum);
// 6

const sumObj = counts.reduce((acc, count) => ({ output: acc.output + count }), {
    output: 0
});
console.log(sumObj);
// { output: 6 }
```

export default Code;

```javascript Softmax
/** e.g.
 * given softmax([1,2,3])
 * returns
 * e_sum = e^1 + e^2 + e^3
 * [e^1 / e_sum, e^2 / e_sum, e^3 / e_sum]
 */
function softmax(inputs) {
    const eSum = inputs.reduce((acc, input) => acc + Math.exp(input));
    return inputs.map(input => Math.exp(input) / eSum);
}

const inputs = [1, 2, 3];
const outputs = softmax(inputs);
console.log(outputs);
```

# ES6

## Arrow Functions

---

export default Code;

```javascript Convert to Arrow Functions
const fn_0 = function() {};
fn_0();

const fn_1 = function(a) {
    console.log(a);
};
fn_1(1);

const fn_2 = function(a, b) {
    return a + b;
};
console.log(fn_2(1, 2));

const fn_3 = function(a) {
    return { a };
};
console.log(fn_3(1));

const fn_4 = function(a) {
    return function(b) {
        return Math.pow(a, b);
    };
};
console.log(fn_4(2)(3));
```
